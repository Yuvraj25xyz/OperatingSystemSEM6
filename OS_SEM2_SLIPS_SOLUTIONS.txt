#include <stdio.h>  

/* Global variables */  

int n, m;  

int Alloc[10][10], Max[10][10], Need[10][10];  

int Avail[10];  

/* Accept Allocation or Max matrix */  

void accept_matrix(int A[10][10])  

{  

    int i, j;  

    for (i = 0; i < n; i++)  

        for (j = 0; j < m; j++)  

            scanf("%d", &A[i][j]);  

}  

  

/* Accept Available vector */  

void accept_available()  

{  



    int i;  

    for (i = 0; i < m; i++)  

        scanf("%d", &Avail[i]);  

}  

  

/* Calculate Need matrix */  

void find_need()  

{  

    int i, j;  

    for (i = 0; i < n; i++)  

        for (j = 0; j < m; j++)  

            Need[i][j] = Max[i][j] - Alloc[i][j];  

}  

  

/* Display Allocation, Max, Need and Available */  

void display_matrix()  

{  

    int i, j;  

    printf("\nProcess\tAllocation\tMax\t\tNeed\n");  

  

    for (i = 0; i < n; i++)  

    {  

        printf("P%d\t", i);  

  

        for (j = 0; j < m; j++)  

            printf("%d ", Alloc[i][j]);  



  

        printf("\t\t");  

  

        for (j = 0; j < m; j++)  

            printf("%d ", Max[i][j]);  

  

        printf("\t\t");  

  

        for (j = 0; j < m; j++)  

            printf("%d ", Need[i][j]);  

  

        printf("\n");  

    }  

  

    printf("\nAvailable:\n");  

    for (j = 0; j < m; j++)  

        printf("%d ", Avail[j]);  

}  

  

/* Main function */  

int main()  

{  

    printf("Enter number of processes and resources: ");  

    scanf("%d %d", &n, &m);  

  

    printf("Enter Allocation matrix:\n");  



    accept_matrix(Alloc);  

  

    printf("Enter Max matrix:\n");  

    accept_matrix(Max);  

  

    printf("Enter Available resources:\n");  

    accept_available();  

  

    find_need();  

    display_matrix();  

    return 0;  

}

Q.1 Write a C program to simulate Banker’s algorithm for the purpose of deadlock

b) Is the system in safe state? If display the safe sequence.

/* Global variables */  

int n, m;  

int Alloc[10][10], Max[10][10], Need[10][10];  

int Avail[10], Work[10];  

int Finish[10], Safe[10];  

int Request[10], proc;  

  

/* Accept matrix */  

void accept_matrix(int A[10][10])  

{  

    int i, j;  

    for(i = 0; i < n; i++)  

        for(j = 0; j < m; j++)  

            scanf("%d", &A[i][j]);

/* Accept Available vector */  

void accept_available()

void find_need()

    int i, j;  

    for(i = 0; i < n; i++)  

        for(j = 0; j < m; j++)  

            Need[i][j] = Max[i][j] - Alloc[i][j];

/* Accept request */  

void accept_request()  

{  

    int i;  

    printf("Enter process number: ");  

    scanf("%d", &proc);  

  

    printf("Enter request:\n");  

    for(i = 0; i < m; i++)  



        scanf("%d", &Request[i]);  

}  

  

/* Safety Algorithm (Simplified) */  

int safety_algo()  

{  

    int i, j, k, count = 0;  

  

    /* Initialize Work = Available */  

    for(i = 0; i < m; i++)  

        Work[i] = Avail[i];  

  

    /* Initialize Finish = false */  

    for(i = 0; i < n; i++)  

        Finish[i] = 0;  

  

    while(count < n)  

    {  

        int allocated = 0;  

  

        for(i = 0; i < n; i++)  

        {  

            if(Finish[i] == 0)  

            {  

                int canExecute = 1;  

  



                for(j = 0; j < m; j++)  

                {  

                    if(Need[i][j] > Work[j])  

                    {  

                        canExecute = 0;  

                        break;  

                    }  

                }  

  

                if(canExecute)  

                {  

                    for(k = 0; k < m; k++)  

                        Work[k] += Alloc[i][k];  

  

                    Safe[count++] = i;  

                    Finish[i] = 1;  

                    allocated = 1;  

                }  

            }  

        }  

  

        if(allocated == 0)  

            return 0;   // Unsafe  

    }  

  

    return 1;   // Safe  



}  

  

/* Resource Request Algorithm */  

void resource_request_algo()  

{  

    int i;  

  

    /* Request <= Need */  

    for(i = 0; i < m; i++)  

    {  

        if(Request[i] > Need[proc][i])  

        {  

            printf("Error: Request exceeds maximum need\n");  

            return;  

        }  

    }  

  

    /* Request <= Available */  

    for(i = 0; i < m; i++)  

    {  

        if(Request[i] > Avail[i])  

        {  

            printf("Resources not available. Process must wait\n");  

            return;  

        }  

    }  



  

    /* Pretend allocation */  

    for(i = 0; i < m; i++)  

    {  

        Avail[i] -= Request[i];  

        Alloc[proc][i] += Request[i];  

        Need[proc][i] -= Request[i];  

    }  

  

    /* Check safety */  

    if(safety_algo())  

    {  

        printf("\nSystem is SAFE\nSafe Sequence: ");  

        for(i = 0; i < n; i++)  

            printf("P%d ", Safe[i]);  

    }  

    else  

    {  

        printf("\nSystem is UNSAFE\n");  

    }  

}  

  

/* Main function */  

int main()  

{  

    printf("Enter number of processes and resources: ");  



    scanf("%d %d", &n, &m);  

  

    printf("Enter Allocation matrix:\n");  

    accept_matrix(Alloc);  

  

    printf("Enter Max matrix:\n");  

    accept_matrix(Max);  

  

    printf("Enter Available resources:\n");  

    accept_available();  

  

    find_need();  

    accept_request();  

    resource_request_algo();  

  

    return 0;  

} 

 

Q2.) Write an MPI program to calculate sum and average of randomly generated 1000 
numbers (stored in array) on a cluster 

 

CODE: 

#include <mpi.h>  

#include <stdio.h>  

#include <stdlib.h>  

#include <time.h>  



  

#define SIZE 1000 // Number of elements  

  

int main(int argc, char* argv[]) {  

    int rank, size;  

    int local_sum = 0, global_sum = 0;  

    double average;  

    int numbers[SIZE];  

  

    MPI_Init(&argc, &argv);  

    MPI_Comm_rank(MPI_COMM_WORLD, &rank);  

    MPI_Comm_size(MPI_COMM_WORLD, &size);  

  

    int chunk_size = SIZE / size;  

    int local_numbers[chunk_size];  

  

    if (rank == 0) {  

        srand(time(NULL));  

        for (int i = 0; i < SIZE; i++) {  

            numbers[i] = rand() % 100; // Random numbers (0-99)  

        }  

    }  

  

    MPI_Scatter(numbers, chunk_size, MPI_INT, local_numbers, chunk_size, MPI_INT, 0,  

MPI_COMM_WORLD);  

  



    for (int i = 0; i < chunk_size; i++) {  

        local_sum += local_numbers[i];  

    }  

  

    MPI_Reduce(&local_sum, &global_sum, 1, MPI_INT, MPI_SUM, 0, 
MPI_COMM_WORLD);  

  

    if (rank == 0) {  

        average = (double)global_sum / SIZE;  

        printf("Total Sum: %d\n", global_sum);  

        printf("Average: %.2f\n", average);  

    }  

  

    MPI_Finalize();  

    return 0;  

}

Q.2 Write a simulation program for disk scheduling using SCAN algorithm. Accept total 
number of disk blocks, disk request string, and current head position from the user. 
Display the list of request in the order in which it is served.  

Also display the total number of head moments. 86, 147, 91, 170, 95, 130, 102, 70 

Starting Head position= 125  

Direction: Left 

 

CODE: 

#include <stdio.h>  

#include <stdlib.h>  

int main() {  

int n, head, direction, totalMovement = 0;  

// Input number of requests  



printf("Enter the number of requests: ");  

scanf("%d", &n);  

int requests[n + 1]; // Add 1 for the head  

int maxTrack;  

// Input maximum track number  

printf("Enter the maximum track number: ");  

scanf("%d", &maxTrack);  

printf("Enter the requests (disk track numbers): ");  

for (int i = 0; i < n; i++) {  

scanf("%d", &requests[i]);  

}  

// Input the initial head position  

printf("Enter the initial head position: ");  

scanf("%d", &head);  

// Input direction (0 for left, 1 for right)  

printf("Enter the direction (0 for left, 1 for right): ");  

scanf("%d", &direction);  

// Add the head to the requests array  

requests[n] = head;  

n++;  

    // Sort the requests array (simple bubble sort for clarity)  

    for (int i = 0; i < n - 1; i++) {  

        for (int j = 0; j < n - i - 1; j++) {  

            if (requests[j] > requests[j + 1]) {  

                int temp = requests[j];  

                requests[j] = requests[j + 1];  



                requests[j + 1] = temp;  

            }  

        }  

    }  

  

    printf("SCAN Disk Scheduling Algorithm\n");  

    printf("Head movement sequence: %d", head);  

  

    // Find the position of the initial head in the sorted array  

    int pos;  

    for (int i = 0; i < n; i++) {  

        if (requests[i] == head) {  

            pos = i;  

            break;  

        }  

    }  

  

    // Move in the chosen direction  

    if (direction == 1) { // Moving right  

        for (int i = pos + 1; i < n; i++) {  

            totalMovement += abs(requests[i] - head);  

            head = requests[i];  

            printf(" -> %d", head);  

        }  

        if (head != maxTrack) { // Go to the end of the track  

            totalMovement += abs(maxTrack - head);  



            head = maxTrack;  

            printf(" -> %d", head);  

        }  

        for (int i = pos - 1; i >= 0; i--) { // Move left  

            totalMovement += abs(requests[i] - head);  

            head = requests[i];  

            printf(" -> %d", head);  

        }  

    } else { // Moving left  

        for (int i = pos - 1; i >= 0; i--) {  

            totalMovement += abs(requests[i] - head);  

            head = requests[i];  

            printf(" -> %d", head);  

        }  

        if (head != 0) { // Go to the start of the track  

            totalMovement += head;  

            head = 0;  

            printf(" -> %d", head);  

        }  

        for (int i = pos + 1; i < n; i++) { // Move right  

            totalMovement += abs(requests[i] - head);  

            head = requests[i];  

            printf(" -> %d", head);  

        }  

    }  

  



    printf("\nTotal head movement: %d\n", totalMovement);  

  

    return 0;  

}

Q.2 Write an MPI program to find the max number from randomly generated 1000 
numbers (stored in array) on a cluster (Hint: Use MPI_Reduce) 

CODE: 

#include <stdio.h>  

#include <stdlib.h>  

#include <mpi.h>  

#define N 1000  

int main(int argc, char *argv[]) {  

int rank, size;  

int numbers[N];  

int local_n;  

int local_min, local_max;  

int global_min, global_max;  

MPI_Init(&argc, &argv);  

MPI_Comm_rank(MPI_COMM_WORLD, &rank);  

MPI_Comm_size(MPI_COMM_WORLD, &size);  

local_n = N / size;  

if(rank == 0) {  

for(int i = 0; i < N; i++) {  

numbers[i] = rand() % 1000;  

}  

}  

int local_numbers[local_n];  

MPI_Scatter(numbers, local_n, MPI_INT,  

local_numbers, local_n, MPI_INT,  

0, MPI_COMM_WORLD);  



local_min = local_numbers[0];  

local_max = local_numbers[0];  

for(int i = 1; i < local_n; i++) {  

if(local_numbers[i] < local_min)  

local_min = local_numbers[i];  

if(local_numbers[i] > local_max)  

local_max = local_numbers[i];  

}  

MPI_Reduce(&local_min, &global_min, 1,  

MPI_INT, MPI_MIN, 0, MPI_COMM_WORLD);  

MPI_Reduce(&local_max, &global_max, 1,  

MPI_INT, MPI_MAX, 0, MPI_COMM_WORLD);  

if(rank == 0) {  

printf("Minimum number = %d\n", global_min);  

printf("Maximum number = %d\n", global_max);  

}  

MPI_Finalize();  

return 0;  

}

                        canExecute = 0;  

                        break;  

                    }  

                }  

  

                if(canExecute)  

                {  

                    for(k = 0; k < m; k++)  

                        Work[k] += Alloc[i][k];  

  

                    Safe[count++] = i;  

                    Finish[i] = 1;  

                    allocated = 1;  

                }  

            }  

        }  

  

        if(allocated == 0)  

            return 0;   // Unsafe  

    }  

  

    return 1;   // Safe  

}  

  

/* Resource Request Algorithm */  

void resource_request_algo()  



{  

    int i;  

  

    /* Request <= Need */  

    for(i = 0; i < m; i++)  

    {  

        if(Request[i] > Need[proc][i])  

        {  

            printf("Error: Request exceeds maximum need\n");  

            return;  

        }  

    }  

  

    /* Request <= Available */  

    for(i = 0; i < m; i++)  

    {  

        if(Request[i] > Avail[i])  

        {  

            printf("Resources not available. Process must wait\n");  

            return;  

        }  

    }  

  

    /* Pretend allocation */  

    for(i = 0; i < m; i++)  

    {  



        Avail[i] -= Request[i];  

        Alloc[proc][i] += Request[i];  

        Need[proc][i] -= Request[i];  

    }  

  

    /* Check safety */  

    if(safety_algo())  

    {  

        printf("\nSystem is SAFE\nSafe Sequence: ");  

        for(i = 0; i < n; i++)  

            printf("P%d ", Safe[i]);  

    }  

    else  

    {  

        printf("\nSystem is UNSAFE\n");  

    }  

}

    int rank, size;  

    int local_sum = 0, global_sum = 0;  

    double average;  

    int numbers[SIZE];  

  

    MPI_Init(&argc, &argv);  

    MPI_Comm_rank(MPI_COMM_WORLD, &rank);  

    MPI_Comm_size(MPI_COMM_WORLD, &size);

    }  

  

    MPI_Scatter(numbers, chunk_size, MPI_INT, local_numbers, chunk_size, MPI_INT, 0,  

MPI_COMM_WORLD);  

  

    for (int i = 0; i < chunk_size; i++) {  

        local_sum += local_numbers[i];  

    }  

  

    MPI_Reduce(&local_sum, &global_sum, 1, MPI_INT, MPI_SUM, 0, 
MPI_COMM_WORLD);  

  

    if (rank == 0) {  

        average = (double)global_sum / SIZE;  

        printf("Total Sum: %d\n", global_sum);  

        printf("Average: %.2f\n", average);  

    }

/* Safety Algorithm (Simplified) */  

int safety_algo()  

{  

    int i, j, k, count = 0;  

  

    /* Initialize Work = Available */  

    for(i = 0; i < m; i++)  

        Work[i] = Avail[i];  

  

    /* Initialize Finish = false */  

    for(i = 0; i < n; i++)  

        Finish[i] = 0;  

  

    while(count < n)  

    {  

        int allocated = 0;  

  

        for(i = 0; i < n; i++)  

        {  

            if(Finish[i] == 0)  

            {  

                int canExecute = 1;  

  

                for(j = 0; j < m; j++)  

                {  



                    if(Need[i][j] > Work[j])  

                    {  

                        canExecute = 0;  

                        break;  

                    }  

                }  

  

                if(canExecute)  

                {  

                    for(k = 0; k < m; k++)  

                        Work[k] += Alloc[i][k];  

  

                    Safe[count++] = i;  

                    Finish[i] = 1;  

                    allocated = 1;  

                }  

            }  

        }  

  

        if(allocated == 0)  

            return 0;   // Unsafe  

    }  

  

    return 1;   // Safe  

}  

  



/* Resource Request Algorithm */  

void resource_request_algo()  

{  

    int i;  

  

    /* Request <= Need */  

    for(i = 0; i < m; i++)  

    {  

        if(Request[i] > Need[proc][i])  

        {  

            printf("Error: Request exceeds maximum need\n");  

            return;  

        }  

    }  

  

    /* Request <= Available */  

    for(i = 0; i < m; i++)  

    {  

        if(Request[i] > Avail[i])  

        {  

            printf("Resources not available. Process must wait\n");  

            return;  

        }  

    }  

  

    /* Pretend allocation */  



    for(i = 0; i < m; i++)  

    {  

        Avail[i] -= Request[i];  

        Alloc[proc][i] += Request[i];  

        Need[proc][i] -= Request[i];  

    }  

  

    /* Check safety */  

    if(safety_algo())  

    {  

        printf("\nSystem is SAFE\nSafe Sequence: ");  

        for(i = 0; i < n; i++)  

            printf("P%d ", Safe[i]);  

    }  

    else  

    {  

        printf("\nSystem is UNSAFE\n");  

    }  

}

    printf("Enter Available resources:\n");  

    accept_available();

void find_need()  

{  

    int i, j;  

    for(i = 0; i < n; i++)  

        for(j = 0; j < m; j++)  

            Need[i][j] = Max[i][j] - Alloc[i][j];  

}  

  

/* Accept request */  

void accept_request()  

{  

    int i;  

    printf("Enter process number: ");  

    scanf("%d", &proc);  

  

    printf("Enter request:\n");  

    for(i = 0; i < m; i++)  

        scanf("%d", &Request[i]);  

}  



  

/* Safety Algorithm (Simplified) */  

int safety_algo()  

{  

    int i, j, k, count = 0;  

  

    /* Initialize Work = Available */  

    for(i = 0; i < m; i++)  

        Work[i] = Avail[i];  

  

    /* Initialize Finish = false */  

    for(i = 0; i < n; i++)  

        Finish[i] = 0;  

  

    while(count < n)  

    {  

        int allocated = 0;  

  

        for(i = 0; i < n; i++)  

        {  

            if(Finish[i] == 0)  

            {  

                int canExecute = 1;  

  

                for(j = 0; j < m; j++)  

                {  



                    if(Need[i][j] > Work[j])  

                    {  

                        canExecute = 0;  

                        break;  

                    }  

                }  

  

                if(canExecute)  

                {  

                    for(k = 0; k < m; k++)  

                        Work[k] += Alloc[i][k];  

  

                    Safe[count++] = i;  

                    Finish[i] = 1;  

                    allocated = 1;  

                }  

            }  

        }  

  

        if(allocated == 0)  

            return 0;   // Unsafe  

    }  

  

    return 1;   // Safe  

}  

  



/* Resource Request Algorithm */  

void resource_request_algo()  

{  

    int i;  

  

    /* Request <= Need */  

    for(i = 0; i < m; i++)  

    {  

        if(Request[i] > Need[proc][i])  

        {  

            printf("Error: Request exceeds maximum need\n");  

            return;  

        }  

    }  

  

    /* Request <= Available */  

    for(i = 0; i < m; i++)  

    {  

        if(Request[i] > Avail[i])  

        {  

            printf("Resources not available. Process must wait\n");  

            return;  

        }  

    }  

  

    /* Pretend allocation */  



    for(i = 0; i < m; i++)  

    {  

        Avail[i] -= Request[i];  

        Alloc[proc][i] += Request[i];  

        Need[proc][i] -= Request[i];  

    }  

  

    /* Check safety */  

    if(safety_algo())  

    {  

        printf("\nSystem is SAFE\nSafe Sequence: ");  

        for(i = 0; i < n; i++)  

            printf("P%d ", Safe[i]);  

    }  

    else  

    {  

        printf("\nSystem is UNSAFE\n");  

    }  

}

    accept_matrix(Max);

break;  

}  

}  

if(direction == 1) {  // Moving Right  

// Move right  

for(int i = pos + 1; i < n; i++) {  

totalMovement += abs(requests[i] - head);  

head = requests[i];  

printf(" -> %d", head);  

}  

// Jump to smallest request  

for(int i = 0; i < pos; i++) {  



totalMovement += abs(requests[i] - head);  

head = requests[i];  

printf(" -> %d", head);  

}  

} else {  // Moving Left  

// Move left  

for(int i = pos - 1; i >= 0; i--) {  

totalMovement += abs(requests[i] - head);  

head = requests[i];  

printf(" -> %d", head);  

}  

// Jump to largest request  

for(int i = n - 1; i > pos; i--) {  

totalMovement += abs(requests[i] - head);  

head = requests[i];  

printf(" -> %d", head);  

}  

}  

printf("\nTotal head movement: %d\n", totalMovement);  

return 0;  

}

#define N 1000

int local_n;  

int local_min, local_max;  

int global_min, global_max;

MPI_Comm_rank(MPI_COMM_WORLD, &rank);  

MPI_Comm_size(MPI_COMM_WORLD, &size);

for(int i = 0; i < N; i++) {  

numbers[i] = rand() % 1000;  

}  

}

local_numbers, local_n, MPI_INT,  



0, MPI_COMM_WORLD);  

local_min = local_numbers[0];  

local_max = local_numbers[0];

local_max = local_numbers[i];

MPI_Reduce(&local_min, &global_min, 1,  

MPI_INT, MPI_MIN, 0, MPI_COMM_WORLD);
